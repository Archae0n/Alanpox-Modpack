# Fun Facilitator
bugged_fun_facilitator = {
	category = specialist
	condition_string = SPECIALIST_JOB_TRIGGER
	building_icon = building_amusement_megaplex
	clothes_texture_index = 3
	
	possible_pre_triggers = {
		has_owner = yes
		is_enslaved = no
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}
	possible = {
		entertainer_job_check_trigger = yes
	}
	
	resources = {
		category = planet_entertainers
		upkeep = {
			consumer_goods = 3
		}
	}

	triggered_planet_modifier = {
		potential = {
			always = yes
		}
		planet_amenities_add = 30
		mult = value:scripted_modifier_mult|MODIFIER|pop_job_amenities_mult|
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_pleasure_seeker = yes }
		}
		modifier = {
			pop_growth_speed = 0.01
		}
	}


	weight = {
		weight = @high_prio_specialist_job_weight
		mult = value:scripted_modifier_job_weight_mult|MODIFIER|pop_job_amenities_mult|
		modifier = {
			factor = 0.2
			has_citizenship_rights = no
			NOT = { has_trait = trait_mechanical }
		}
		modifier = {
			factor = 3
			has_trait = trait_robot_domestic_protocols
			OR = {
				is_non_sapient_robot = yes
				is_shackled_robot = yes
			}
		}
		modifier = {
			factor = 1.5
			has_job = bugged_fun_facilitator
		}
		modifier = {
			factor = 0.1
			is_domestic_servant = yes
		}
		modifier = {
			factor = 0.1
			NOT = { has_job = bugged_fun_facilitator }
			exists = planet
			planet = {
				free_amenities > 5
			}
		}
		modifier = {
			factor = 2
			NOT = { has_job = bugged_fun_facilitator }
			exists = planet
			planet = {
				free_amenities < 2
			}
		}
	}
}

	category = criminal
	can_set_priority = no
	contributes_to_diplo_weight = no
	condition_string = CRIMINAL_JOB_TRIGGER
	building_icon = building_bio_reprocessing_facilities
	clothes_texture_index = 2

	possible_pre_triggers = {
		has_owner = yes
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}

	possible = {
		# check that this is not a robot in servitude
		NOT = {
			has_citizenship_type = {
				country = owner
				type = citizenship_robot_servitude
			}
		}
		owner = {
			is_gestalt = no
		}
	}

	resources = {
		category = planet_farmers
		produces = {
			trigger = {
				criminal_positives = yes
			}
			food = 6
		}
		produces = {
			trigger = {
				exists = owner
				owner = {
					AND = {
						criminal_positives = yes
						is_robot_empire = yes
					}
				}
			}
			food = -1
		}
		produces = {
			trigger = {
				AND = {
					criminal_positives = yes
					planet = { has_active_building = building_food_processing_facility  }
				}
			}
			food = 1
		}
		produces = {
			trigger = {
				AND = {
					criminal_positives = yes
					planet = { has_active_building = building_food_processing_center }
				}
			}
			food = 2
		}
		upkeep = {
			trigger = {
				exists = owner
				owner = {
					has_edict = farming_subsidies
				}
			}
			energy = 0.5
		}
		produces = {
			trigger = {
				AND = {
					criminal_positives = yes
					owner = {
						OR = {
							has_valid_civic = civic_philosopher_king
							has_valid_civic = civic_bugged_utopian_democracy
							has_valid_civic = civic_bugged_philosophers_assembly
							has_valid_civic = civic_bugged_enlightened_bureaucracy
						}
					}
				}
			}
			unity = 0.1
		}
	}

	
	weight = {
		weight = @worker_job_weight
		mult = value:job_weights_modifier|JOB|farmer|RESOURCE|food|
		modifier = {
			factor = 1.1
			owner = { has_valid_civic = civic_agrarian_idyll }
		}
		modifier = {
			factor = 10
			is_enslaved = yes
			can_take_servant_job = no
			NOT = { has_slavery_type = { type = slavery_indentured } }
		}
		modifier = {
			factor = 3
			OR = {
				is_non_sapient_robot = yes
				is_shackled_robot = yes
			}
			can_take_servant_job = no
			owner = { has_technology = tech_droid_workers }
		}
		modifier = {
			factor = 200
			OR = {
				is_non_sapient_robot = yes
				is_shackled_robot = yes
			}
			can_take_servant_job = no
			owner = { NOT = { has_technology = tech_droid_workers } }
		}
		modifier = {
			factor = 0.25
			can_take_servant_job = yes
		}
		# low income modifier
		modifier = {
			factor = 1.2
			exists = planet
			planet = {
				has_available_jobs = farmer
			}
			exists = owner
			owner = {
				is_ai = yes
				has_monthly_income = {
					resource = food
					value < 10
				}
			}
		}

		modifier = {
			factor = 0.5
			has_trait = trait_lithoid

			owner = {
				is_lithoid_empire = yes
				has_monthly_income = {
					resource = food
					value >= 0
				}
			}
		}
	}
}
